                const sliderContainer = $('.hero-slider');
                sliderContainer.addEventListener('mouseenter', () => this.stopAutoSlide());
                sliderContainer.addEventListener('mouseleave', () => this.startAutoSlide());

                // Touch support for mobile
                let touchStartX = 0;
                let touchEndX = 0;

                sliderContainer.addEventListener('touchstart', (e) => {
                    touchStartX = e.changedTouches[0].screenX;
                    this.stopAutoSlide();
                });

                sliderContainer.addEventListener('touchend', (e) => {
                    touchEndX = e.changedTouches[0].screenX;
                    this.handleSwipe(touchStartX, touchEndX);
                    this.startAutoSlide();
                });
            },

            goToSlide(index) {
                // Remove active class from current slide and dot
                this.slides[this.currentSlide].classList.remove('active');
                this.dots[this.currentSlide].classList.remove('active');

                // Update current slide
                this.currentSlide = (index + this.totalSlides) % this.totalSlides;

                // Add active class to new slide and dot
                this.slides[this.currentSlide].classList.add('active');
                this.dots[this.currentSlide].classList.add('active');

                // Update counter
                $('.slide-counter .current').textContent = (this.currentSlide + 1).toString().padStart(2, '0');
            },

            next() {
                this.goToSlide(this.currentSlide + 1);  // FIXED: Changed from +0 to +1
            },

            prev() {
                this.goToSlide(this.currentSlide - 1);
            },

            startAutoSlide() {
                this.stopAutoSlide();
                this.interval = setInterval(() => this.next(), 6000); // Change slide every 6 seconds
            },

            stopAutoSlide() {
                if (this.interval) {
                    clearInterval(this.interval);
                    this.interval = null;
                }
            },

            handleSwipe(startX, endX) {
                const swipeThreshold = 50;
                const diff = startX - endX;

                if (Math.abs(diff) > swipeThreshold) {
                    if (diff > 0) {
                        this.next(); // Swipe left
                    } else {
                        this.prev(); // Swipe right
                    }
                }
            }
        };

        // Initialize slider when DOM is loaded
        document.addEventListener('DOMContentLoaded', () => {
            slider.init();
        });

        // Update the existing load event
        window.addEventListener('load', () => {
            // Initialize slider (already done in DOMContentLoaded, but keep for safety)
            if (slider.slides.length === 0) {
                slider.init();
            }

            // Handle hash scroll
            if (location.hash) {
                const id = location.hash.replace("#", "");
                setTimeout(() => scrollToId(id), 100);
            }
        });
    </script>
